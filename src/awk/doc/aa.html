function readcsv(f,_Tbl, str,row,head) {
  FS=","; RS="\n"
  while (str = line(f)) {
    if (str == -1) return 
    if (str ~ /^[ \t]*$/) continue
    split(str,a,",")
    if (row++)
        cells(a,row,_Tbl)
    else
        header0(a,_Tbl)
}
